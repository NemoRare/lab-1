#include"stdio.h"
#include "stdafx.h"
#include "time.h"
#include "locale.h"
#include "stdlib.h"
#define SIZE 10
int main(void) {
	setlocale(LC_ALL,"RUS");
    int array[SIZE]; 
    srand((unsigned)time(NULL));
    for (int i = 0; i < SIZE; i++) {
        array[i] = rand() % 100; 
    }

    printf("Случайные числа в массиве:\n");
    for (int i = 0; i < SIZE; i++) {
        printf("%d ", array[i]);
    }
    printf("\n");

  system("pause"); 
}
